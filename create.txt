--CREATE TABLES

CREATE TABLE AUTHOR(
AUTHOR_ID      NUMBER(5),
A_LASTNAME     VARCHAR2(15),
A_FIRSTNAME    VARCHAR2(12),
DATE_OF_BIRTH  DATE,
DATE_OF_DEATH  DATE,
COUNTRY        VARCHAR2(30) DEFAULT 'GREECE',
GENDER         CHAR DEFAULT 'F'  --SAME AS CHAR(1)
);

CREATE TABLE BOOKS(
ISBN NUMBER(13),
TITLE VARCHAR2(50),
AUTHOR_ID NUMBER(5),
GENRE VARCHAR2(20) DEFAULT 'OTHER',
EDITION NUMBER(2),
PAGES NUMBER(5),
DATE_PUBLISHED DATE,
PRICE NUMBER(6,2)
);

CREATE TABLE COPY(
COPY_ID           VARCHAR2(16),
ISBN              NUMBER(13),
DATE_OF_PURCHASE  DATE
);

CREATE TABLE USERS(
USER_ID         NUMBER(4),
US_LASTNAME     VARCHAR2(30),
US_FIRSTNAME    VARCHAR2(30),
EMAIL           VARCHAR2(20),
CONTACT_NUMBER  NUMBER(15),
GENDER          CHAR DEFAULT 'F' --SAME AS CHAR(1),
);

CREATE TABLE LOANS(
LOAN_ID      NUMBER(5),
USER_ID      NUMBER(4),
COPY_ID      VARCHAR2(16),
LOAN_DATE    DATE,
RETURN_DATE  DATE
);


--CREATE SEQUENCES

MAXVALUE 100000
